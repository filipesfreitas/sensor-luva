97 class BinStruct(object):
98  """Binary structure representation
99 
100  Subclasses must specify actual structure layout using 'fields' and 'format' members.
101  For example, the following subclass represents structure with two fields:
102  f1 of size 2 bytes and 4 bytes f2. Little endian.
103  class SomeStruct(BinStruct):
104  fields = ("f1",
105  "f2")
106  format = "<HL"
107 
108  Then subclass can be used to initialize fields of underlaying structure and convert it to binary representation:
109  f = open('some_struct.bin', 'wb')
110  s = SomeStruct()
111  s.f1 = 1
112  s.f2 = 10
113  f.write(s.dump())
114  f.close()
115  """

118  """Base constructor for binary structure objects
119  """
120  if buf is None:
121  buf = b'\0' * self.sizeofsizeof()
122  fields = struct.unpack(self.__class__.format, buf[:self.sizeofsizeof()])
123  self.__dict__.update(zip(self.__class__.fields, fields))

604  # {

630  # XT_SOL_EXIT = 0

632  XT_SOL_PS = 2
