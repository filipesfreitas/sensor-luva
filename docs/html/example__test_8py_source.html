24 def udp_client(address, payload):
25  for res in socket.getaddrinfo(address, PORT, socket.AF_UNSPEC,
26  socket.SOCK_DGRAM, 0, socket.AI_PASSIVE):
27  family_addr, socktype, proto, canonname, addr = res
28  try:
29  sock = socket.socket(family_addr, socket.SOCK_DGRAM)
30  except socket.error as msg:
31  print('Could not create socket: ' + str(msg[0]) + ': ' + msg[1])
32  raise
33  try:
34  sock.sendto(payload, addr)
35  reply, addr = sock.recvfrom(128)
36  if not reply:
37  return
38  print('Reply[' + addr[0] + ':' + str(addr[1]) + '] - ' + str(reply))
39  except socket.error as msg:
40  print('Error Code : ' + str(msg[0]) + ' Message: ' + msg[1])
41  sock.close()
42  raise
43  return reply

46 @ttfw_idf.idf_example_test(env_tag="Example_WIFI")

59  ttfw_idf.log_performance("udp_server_bin_size", "{}KB".format(bin_size // 1024))
60  ttfw_idf.check_performance("udp_server_bin_size", bin_size // 1024, dut1.TARGET)
61 
62  # start test
63  dut1.start_app()
64 
65  ipv4 = dut1.expect(re.compile(r" IPv4 address: ([0-9]+\.[0-9]+\.[0-9]+\.[0-9]+)"), timeout=30)[0]
66  ipv6_r = r':'.join((r'[0-9a-fA-F]{4}',) * 8) # expect all 8 octets from IPv6 (assumes it's printed in the long form)
67  ipv6 = dut1.expect(re.compile(r' IPv6 address: ({})'.format(ipv6_r)), timeout=30)[0]
68  print("Connected with IPv4={} and IPv6={}".format(ipv4, ipv6))
