286  ESP_LOGE(TAG, "failed to post on queue");
287  }

292  , ret == ESP_OK ? "SENSOR 0X53" : ret1 != ESP_OK ? "SENSOR 0X1D" : "None");
293  }
294  xEventGroupSync(xEventGroup,(EventBits_t) pvParameters,ALLSYNCH,portMAX_DELAY);
295 
296  }

305 static void udp_server_task(void *pvParameters)
306 {
307  char error_code_[512];
308  char addr_str[128];
309  int addr_family = (int)pvParameters;
310  int ip_protocol = 0;
311  struct sockaddr_in6 dest_addr;
312 
313  char rx_buffer[128]={'0'};
314  int len_to_send=0;
315  int16_t tx_buffer[7];
316  char tx_buffer_msg[128]={'0'};
317  int i;
318 
319  if (addr_family == AF_INET) {
320  struct sockaddr_in *dest_addr_ip4 = (struct sockaddr_in *)&dest_addr;
321  dest_addr_ip4->sin_addr.s_addr = htonl(INADDR_ANY);
322  dest_addr_ip4->sin_family = AF_INET;
323  dest_addr_ip4->sin_port = htons(PORT);
324  ip_protocol = IPPROTO_IP;
325  } else if (addr_family == AF_INET6) {
326  bzero(&dest_addr.sin6_addr.un, sizeof(dest_addr.sin6_addr.un));
327  dest_addr.sin6_family = AF_INET6;
328  dest_addr.sin6_port = htons(PORT);
329  ip_protocol = IPPROTO_IPV6;
330  }
331  while(1){
332  int sock = socket(addr_family, SOCK_DGRAM, ip_protocol);
333  if (sock < 0) {
334  ESP_LOGE(TAG, "Unable to create socket: errno %d", errno);
335  break;
336  }
337  ESP_LOGI(TAG, "Socket created");
338 
339 
340 
341  int err = bind(sock, (struct sockaddr *)&dest_addr, sizeof(dest_addr));
342  if (err < 0) {
343  ESP_LOGE(TAG, "Socket unable to bind: errno %d", errno);
344  }
345  ESP_LOGI(TAG, "Socket bound, port %d", PORT);
346 
347  ESP_LOGI(TAG, "Waiting for data");
348  struct sockaddr_in6 source_addr; // Large enough for both IPv4 or IPv6
349  socklen_t socklen = sizeof(source_addr);
350  int len = recvfrom(sock, rx_buffer, sizeof(rx_buffer) - 1, 0, (struct sockaddr *)&source_addr, &socklen);
351 
352  // Error occurred during receiving
353  if (len < 0) {
354  ESP_LOGE(TAG, "recvfrom failed: errno %d", errno);
355  break;
356  }
357  // Data received
358  else {
359 
360  // Get the sender's ip address as string
361  if (source_addr.sin6_family == PF_INET) {
362  inet_ntoa_r(((struct sockaddr_in *)&source_addr)->sin_addr.s_addr, addr_str, sizeof(addr_str) - 1);
363  } else if (source_addr.sin6_family == PF_INET6) {
364  inet6_ntoa_r(source_addr.sin6_addr, addr_str, sizeof(addr_str) - 1);
365  }
366 
367  rx_buffer[len] = 0; // Null-terminate whatever we received and treat like a string...
368  ESP_LOGI(TAG, "Received %d bytes from %s:", len, addr_str);
369  ESP_LOGI(TAG, "%s", rx_buffer);
370 
371  while(1){
372  xEventGroupWaitBits(xEventGroup, PORT0ADX | PORT1ADX, pdFALSE,pdTRUE,(TickType_t)1);
373  i =1;
374  do{
375  if(xQueueReceive(buffer_queue, &tx_buffer, pdMS_TO_TICKS(10))==true){
376  len_to_send = sprintf(tx_buffer_msg,"%d;%d;%d;%d;%d;%d;%d\r\n",
377  tx_buffer[0],
378  tx_buffer[1],
379  tx_buffer[2],
380  tx_buffer[3],
381  tx_buffer[4],
382  tx_buffer[5],
383  tx_buffer[6]);
384 
385  int err = sendto(sock, tx_buffer_msg, len_to_send, 0, (struct sockaddr *)&source_addr, sizeof(source_addr));
386 
387  if (err < 0) {
388  ESP_LOGE(TAG, "Error occured during sending: errno %x\t%s", errno,esp_err_to_name_r(errno,error_code_,512));
389  break;
390  }
391  }
392  }while(i++ == 2);
393  if (err<0)
394  {
395  break;
396  }
397  xEventGroupSetBits(xEventGroup,UDP);
398  }
399 }
400 
401 if (sock != -1) {
402  ESP_LOGE(TAG, "Shutting down socket and restarting...");
403  shutdown(sock, 0);
404  close(sock);
405 }
406 }
407 vTaskDelete(NULL);
408 }
